\begin{frame}
	\frametitle{Estimating $S(s)$}
	\begin{itemize}[<+->]
		\item
		\begin{equation} 
			G_1(s) = G_J(s)S(s)
		\end{equation}
		\item
		\begin{equation}
			G_J(s) = \frac{1}{2Hs+K_d}
		\end{equation}
		\item
		\begin{equation}
			2H>>K_d
		\end{equation}
		\item
		\begin{equation}
			S(s) \approx 2HsG_1(s)
		\end{equation}
		\item Need to estimate $H$
	\end{itemize}
\end{frame}
